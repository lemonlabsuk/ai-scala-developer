              /**
                * This method parses a string into a list of CsvValues
                * by splitting on commas
                */
              def parseStringFromString(csv: CSV): List[String] = {
                val left = csv.split(",")
                val right = csv.split(",")
                val empty = if (empty == "" && csv.length() == 2) empty.trim else Nil
                if (left.length() == right.length()) {
                  var i = 0
                  for (k <- csv.split(",")) {
                    if (lsu(k).nonEmpty) {
                      val sv = pw(k)
                      left.drop(i) ++= sv.drop(i)
                      i += 1
                    }
                  }
                  if (right.length() == empty.length()) {
                    left ++= right
                  } else if (left.length() == empty.length()) {
                    left ++= right
                  } else {
                    left ++= empty.drop(i)
                  }
                } else {
                  Nil
                }
              }
